A UI Lib for React Native

## Start

### 1. Install

```js
npm install rnx-ui --save
```

### 2. Use

```js
import React, {
  Component,
} from 'react';
import Btn from 'rnx-ui/Btn';

function Demo(props) {
  return <Btn />;
}
```
